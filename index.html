Código

Despliega en minutos,
no en días

Solo conecta tu repositorio. Nada más.

# Solo conecta tu repo en el dashboard
# TrackCloud detecta automáticamente:

framework: "Next.js 14"
runtime: "Node 20"
build_command: "npm run build"
start_command: "npm start"
port: 3000

# ✅ Deploy exitoso en 47 segundos
FROM node:20-alpine
WORKDIR /app
COPY package*.json ./
RUN npm ci --production
COPY . .
EXPOSE 3000
CMD ["node", "server.js"]

# TrackCloud detecta Dockerfile automáticamente
# Instala el CLI
npm install -g @trackcloud/cli

# Autentícate
trackcloud login

# Deploy desde cualquier directorio
trackcloud up
